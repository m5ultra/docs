var objects = [
  { 'x': 1, 'y': 2 },
  { 'x': 2, 'y': 1 },
  { 'x': 1, 'y': 2 }];

_.uniqWith(objects, _.isEqual);
// => [{ 'x': 1, 'y': 2 }, { 'x': 2, 'y': 1 }]

深层嵌套数据判断是否相等

TIP

使用场景： 页面有好多表单数据 判断用户是否修改 如果有修改，提交按钮可以点击；如果没有修改，按钮置灰

var object = { 'a': 1 };
var other = { 'a': 1 };

_.eq(object, object);
// => true

_.eq(object, other);
// => false

_.eq('a', 'a');
// => true

_.eq('a', Object('a'));
// => false

_.eq(NaN, NaN);
// => true
